# Spark 2.3.2 引擎适配 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 让 BiliAnalysis 的 Spark 分析引擎通过子进程隔离方式在 Spark 2.3.2 上完整运行（Python 3.13 主进程 + Python 3.7 子进程），4 个分析步骤全部可用。

**Architecture:** `SparkBridge`（Python 3.13 侧，实现 `AnalysisEngine`）通过 `subprocess.run` 调起 Python 3.7 子进程中的 `entry.py` → `SparkEngine`。数据通过 stdin/stdout JSON + 文件系统 Parquet 交换。

**Tech Stack:** Python 3.13, Python 3.7, PySpark 2.3.2, subprocess, Pydantic, asyncio

```python
def _extract_tables(self, raw_df: DataFrame) -> dict[str, DataFrame]:
    """从 raw Spark DataFrame 拆出 5 张表，每行带 row_id 用于关联。"""
    # Weekly
    weekly = raw_df.select(
        col("number"),
        col("config.subject").cast("string").alias("subject"),
        col("config.name").cast("string").alias("name"),
        col("config.stime").cast("long").alias("start_time"),
        col("config.etime").cast("long").alias("end_time"),
    )

    # Videos — explode 数组，每个视频一行，分配统一 row_id
    video_rows = raw_df.select(
        col("number").alias("week_number"),
        explode("videos").alias("v")
    )
    video = video_rows.select(
        col("v.aid").alias("aid"),
        col("v.bvid").alias("bvid"),
        col("v.title").alias("title"),
        col("v.desc").alias("desc"),
        col("v.duration").alias("duration"),
        col("v.pubdate").alias("pubdate"),
        col("v.cid").alias("cid"),
        col("v.pic").alias("pic"),
        monotonically_increasing_id().alias("row_id"),
    )

    # Creator / Category / VideoStat 通过 join video.row_id 获得相同 ID
    creator_base = video_rows.select(
        col("v.owner.mid").alias("mid"),
        col("v.owner.name").alias("name"),
        col("v.owner.face").alias("face"),
        monotonically_increasing_id().alias("_join_key"),
    )
    creator = video.select("row_id").withColumn("_join_key", monotonically_increasing_id()) \
        .join(creator_base, "_join_key", "inner") \
        .select("mid", "name", "face", "row_id")

    v_fields = {f.name for f in video_rows.schema["v"].dataType.fields}
    if "rcmd_reason" in v_fields:
        cat_base = video_rows.select(
            col("v.tid").alias("tid"),
            col("v.tname").alias("tname"),
            col("v.rcmd_reason.tid_v2").alias("tid_v2"),
            col("v.rcmd_reason.tname_v2").alias("tname_v2"),
            monotonically_increasing_id().alias("_join_key"),

    else:
        cat_base = video_rows.select(
            col("v.tid").alias("tid"),
            col("v.tname").alias("tname"),
            lit(None).cast(LongType()).alias("tid_v2"),
            lit(None).cast("string").alias("tname_v2"),
            monotonically_increasing_id().alias("_join_key"),
        )

        .join(cat_base, "_join_key", "inner") \
        .select("tid", "tname", "tid_v2", "tname_v2", "row_id")

    stat_base = video_rows.select(
        col("v.stat.aid").alias("aid"),
        col("v.stat.view").alias("view"),
        col("v.stat.like").alias("like"),
        col("v.stat.coin").alias("coin"),

        .select("aid", "view", "like", "coin", "favorite", "share", "reply", "danmaku", "row_id")
```

```python
"""Spark 引擎子进程入口。

stdin 读取 JSON 请求，执行指定分析步骤，stdout 输出 JSON 结果。

用法:
    python3.7 -m bilianalysis.engine.spark.entry < request.json

请求格式:
    {"step": "statistics", "raw_dir": "...", "processed_dir": "...",
     "reports_dir": "...", "spark_master": "local[*]"}

响应格式:
    成功: {"ok": true, "data": {...}}
    失败: {"ok": false, "error": "..."}
"""
import sys
import json
import asyncio
import traceback

from bilianalysis.engine.spark.engine import SparkEngine
from bilianalysis.config.model import DataSection


def main():
    req = json.loads(sys.stdin.read())

    data_config = DataSection(
        raw_dir=req["raw_dir"],
        processed_dir=req["processed_dir"],
        reports_dir=req["reports_dir"],
    )
    spark_master = req.get("spark_master", "local[*]")
    engine = SparkEngine(data_config, spark_master=spark_master)

    step = req["step"]
    method = getattr(engine, step)

    if asyncio.iscoroutinefunction(method):
        result = asyncio.run(method())
    else:
        result = method()

    # Pydantic v2 uses model_dump; fallback to dict() for older versions
    if hasattr(result, "model_dump"):
        data = result.model_dump()
    elif hasattr(result, "dict"):
        data = result.dict()
    else:
        data = result

    print(json.dumps({"ok": True, "data": data}, default=str))


if __name__ == "__main__":
    try:
        main()
    except Exception:
        print(json.dumps({"ok": False, "error": traceback.format_exc()}))
        sys.exit(1)
```

### Task 4: 创建 bridge.py 桥接层

**Files:**
- Create: `src/bilianalysis/engine/spark/bridge.py`

- [ ] **Step 1: 创建 bridge.py**

```python
"""SparkBridge — 子进程桥接层。

实现 AnalysisEngine，每个方法调起 Python 3.7 子进程执行 Spark 分析。
运行在 Python 3.13 主进程中，不 import PySpark。
"""
import asyncio
import json
import subprocess

from bilianalysis.config.model import DataSection, SparkConfig
from bilianalysis.engine.base import (
    AnalysisEngine,
    CleanReport,
    StatReport,
    ClusterReport,
    PredictionReport,
    ModelComparisonReport,
)
from bilianalysis.engine.spark import SparkStepError, SparkRuntimeError


class SparkBridge(AnalysisEngine):
    """子进程桥接引擎——在 Python 3.7 子进程中运行 Spark 2.3.2 分析。"""

    def __init__(self, data_config: DataSection, spark_config: SparkConfig):
        self._raw_dir = str(data_config.raw_dir)
        self._processed_dir = str(data_config.processed_dir)
        self._reports_dir = str(data_config.reports_dir)
        self._py37 = spark_config.python
        self._master = spark_config.master
        self._timeout = spark_config.timeout

    # ── subprocess ─────────────────────────────────────────

    def _run_step(self, step: str) -> dict:
        """调起 py37 子进程，传入 step 名 + 配置，返回 parsed JSON data 字典。"""
        req = {
            "step": step,
            "raw_dir": self._raw_dir,
            "processed_dir": self._processed_dir,
            "reports_dir": self._reports_dir,
            "spark_master": self._master,
        }
        try:
            proc = subprocess.run(
                [self._py37, "-m", "bilianalysis.engine.spark.entry"],
                input=json.dumps(req),
                capture_output=True,
                text=True,
                timeout=self._timeout,
            )
        except FileNotFoundError:
            raise SparkRuntimeError(
                f"Python 3.7 not found at '{self._py37}'. "
                f"Set spark.python in config.yaml to your py37 venv path."
            )
        except subprocess.TimeoutExpired:
            raise SparkStepError(step, f"Timed out after {self._timeout}s")

        # Non-zero exit → stderr has the traceback
        if proc.returncode != 0:
            raise SparkStepError(step, proc.stderr.strip() or f"exit code {proc.returncode}")

        # Parse stdout JSON
        try:
            result = json.loads(proc.stdout)
        except json.JSONDecodeError:
            preview = proc.stdout[:500] if proc.stdout else "(empty stdout)"
            raise SparkStepError(step, f"Invalid JSON from subprocess:\n{preview}")

        if not result.get("ok"):
            raise SparkStepError(step, result.get("error", "unknown error"))

        return result["data"]

    # ── AnalysisEngine 接口 ──────────────────────────────

    async def clean_data(self) -> CleanReport:
        loop = asyncio.get_event_loop()
        result = await loop.run_in_executor(None, self._run_step, "clean_data")
        return CleanReport(**result)

    def statistics(self) -> StatReport:
        result = self._run_step("statistics")
        return StatReport(**result)

    def clustering(self) -> ClusterReport:
        result = self._run_step("clustering")
        return ClusterReport(**result)

    def prediction(self) -> PredictionReport:
        result = self._run_step("prediction")
        return PredictionReport(**result)

    def model_comparison(self) -> ModelComparisonReport:
        raise NotImplementedError(
            "model_comparison 依赖 sklearn 多模型交叉验证，SparkEngine 暂不支持。"
            "请使用 PandasEngine (analysis.engine: pandas)。"
        )
```

```python
class TestSparkBridgeSubprocess:
    """测试 _run_step 子进程交互逻辑。"""

    def test_run_step_success(self, bridge):
        """成功 case：subprocess stdout 返回 valid JSON。"""
        mock_proc = mock.MagicMock()
        mock_proc.stdout = json.dumps({
            "ok": True,
            "data": {
                "total_weeks": 10, "total_videos": 100,
                "duplicates_dropped": 0, "missing_filled": 5,
                "outliers_flagged": 1, "duration_seconds": 12.5,
            },
        })
        mock_proc.returncode = 0

        with mock.patch("subprocess.run", return_value=mock_proc) as mock_run:
            result = bridge._run_step("clean_data")

        assert result["total_weeks"] == 10
        assert result["total_videos"] == 100
        # Verify correct args passed to subprocess
        call_args = mock_run.call_args
        assert call_args[0][0][0] == "python3.7"
        assert "-m" in call_args[0][0]
        sent_req = json.loads(call_args[1]["input"])
        assert sent_req["step"] == "clean_data"
        assert sent_req["spark_master"] == "local[*]"

    def test_run_step_subprocess_failure(self, bridge):
        """子进程崩溃：returncode != 0，stderr 有内容。"""
        mock_proc = mock.MagicMock()
        mock_proc.stdout = ""
        mock_proc.stderr = "Py4JJavaError: Connection refused"
        mock_proc.returncode = 1

        with mock.patch("subprocess.run", return_value=mock_proc):
            with pytest.raises(SparkStepError, match="clean_data.*failed"):
                bridge._run_step("clean_data")

    def test_run_step_json_ok_false(self, bridge):
        """业务异常：JSON 返回 ok: false。"""
        mock_proc = mock.MagicMock()
        mock_proc.stdout = json.dumps({
            "ok": False,
            "error": "ValueError: Invalid aid field",
        })
        mock_proc.returncode = 0

        with mock.patch("subprocess.run", return_value=mock_proc):
            with pytest.raises(SparkStepError, match="Invalid aid field"):
                bridge._run_step("statistics")

    def test_run_step_invalid_json(self, bridge):
        """stdout 不是有效 JSON。"""
        mock_proc = mock.MagicMock()
        mock_proc.stdout = "some garbage output"
        mock_proc.returncode = 0

        with mock.patch("subprocess.run", return_value=mock_proc):
            with pytest.raises(SparkStepError, match="Invalid JSON"):
                bridge._run_step("clustering")

    def test_run_step_python_not_found(self, bridge):
        """Python 3.7 解释器不存在。"""
        with mock.patch("subprocess.run", side_effect=FileNotFoundError):
            with pytest.raises(SparkRuntimeError, match="not found"):
                bridge._run_step("prediction")

    def test_run_step_timeout(self, bridge):
        """子进程超时。"""
        with mock.patch("subprocess.run", side_effect=subprocess.TimeoutExpired(cmd="python3.7", timeout=30)):
            with pytest.raises(SparkStepError, match="Timed out"):
                bridge._run_step("prediction")
```

```python
class TestSparkConfig:
    """测试 Spark 配置节的加载与默认值。"""
    import tempfile, os
    from pathlib import Path

    def test_spark_config_defaults(self, tmp_path):
        from bilianalysis.config.loader import load_config
        config_path = tmp_path / "config.yaml"
        config_path.write_text("analysis:\n  engine: spark\n", encoding="utf-8")
        config = load_config(str(config_path))
        assert config.spark.python == ".venv-spark37/bin/python"
        assert config.spark.master == "local[*]"
        assert config.spark.timeout == 600

    def test_spark_config_from_yaml(self, tmp_path):
        from bilianalysis.config.loader import load_config
        config_path = tmp_path / "config.yaml"
        config_path.write_text("""
analysis:
  engine: spark
spark:
  python: "/opt/spark37/bin/python"
  master: "spark://master:7077"
  timeout: 300
""", encoding="utf-8")
        config = load_config(str(config_path))
        assert config.spark.python == "/opt/spark37/bin/python"
        assert config.spark.master == "spark://master:7077"
        assert config.spark.timeout == 300

    def test_spark_config_env_override(self, tmp_path, monkeypatch):
        from bilianalysis.config.loader import load_config
        monkeypatch.setenv("BILI_SPARK__MASTER", "spark://prod:7077")
        config_path = tmp_path / "config.yaml"
        config_path.write_text("analysis:\n  engine: spark\n", encoding="utf-8")
        config = load_config(str(config_path))
        assert config.spark.master == "spark://prod:7077"
```
